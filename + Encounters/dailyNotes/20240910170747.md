import { generateObject } from 'ai';
import { z } from 'zod';

// 模拟 AI 模型
jest.mock('ai', () => ({
  generateObject: jest.fn(),
}));

describe('Recipe Generator', () => {
  const recipeSchema = z.object({
    name: z.string(),
    ingredients: z.array(z.object({ name: z.string(), amount: z.string() })),
    steps: z.array(z.string()),
  });

  beforeEach(() => {
    jest.clearAllMocks();
  });

  test('成功生成食谱', async () => {
    const mockRecipe = {
      name: '美味烤宽面',
      ingredients: [
        { name: '宽面', amount: '500克' },
        { name: '番茄酱', amount: '400克' },
        { name: '牛肉馅', amount: '300克' },
      ],
      steps: [
        '煮宽面至半熟',
        '炒熟牛肉馅',
        '铺层：面-肉-酱，重复',
        '烤箱200度烤30分钟',
      ],
    };

    (generateObject as jest.Mock).mockResolvedValue({ object: mockRecipe });

    const result = await generateObject({
      model: 'mockModel',
      schema: z.object({ recipe: recipeSchema }),
      prompt: '生成一个烤宽面食谱',
    });

    expect(result.object).toEqual({ recipe: mockRecipe });
    expect(generateObject).toHaveBeenCalledWith({
      model: 'mockModel',
      schema: expect.any(Object),
      prompt: '生成一个烤宽面食谱',
    });
  });

  test('使用 schemaName 和 schemaDescription', async () => {
    const mockRecipe = {
      name: '简易沙拉',
      ingredients: [
        { name: '生菜', amount: '1颗' },
        { name: '番茄', amount: '2个' },
        { name: '黄瓜', amount: '1根' },
      ],
      steps: [
        '洗净并切碎生菜',
        '切片番茄和黄瓜',
        '混合所有食材',
        '淋上喜欢的沙拉酱',
      ],
    };

    (generateObject as jest.Mock).mockResolvedValue({ object: mockRecipe });

    const result = await generateObject({
      model: 'mockModel',
      schemaName: 'Recipe',
      schemaDescription: '一道菜的食谱',
      schema: recipeSchema,
      prompt: '生成一个简单的沙拉食谱',
    });

    expect(result.object).toEqual(mockRecipe);
    expect(generateObject).toHaveBeenCalledWith({
      model: 'mockModel',
      schemaName: 'Recipe',
      schemaDescription: '一道菜的食谱',
      schema: expect.any(Object),
      prompt: '生成一个简单的沙拉食谱',
    });
  });

  test('使用不同的生成模式', async () => {
    const mockRecipe = {
      name: '煎蛋',
      ingredients: [
        { name: '鸡蛋', amount: '2个' },
        { name: '盐', amount: '少许' },
      ],
      steps: [
        '打蛋到碗里',
        '加入少许盐',
        '平底锅加热',
        '倒入蛋液煎熟',
      ],
    };

    (generateObject as jest.Mock).mockResolvedValue({ object: mockRecipe });

    const result = await generateObject({
      model: 'mockModel',
      schema: recipeSchema,
      prompt: '生成一个煎蛋食谱',
      mode: 'json',
    });

    expect(result.object).toEqual(mockRecipe);
    expect(generateObject).toHaveBeenCalledWith({
      model: 'mockModel',
      schema: expect.any(Object),
      prompt: '生成一个煎蛋食谱',
      mode: 'json',
    });
  });

  test('处理生成失败的情况', async () => {
    (generateObject as jest.Mock).mockRejectedValue(new Error('生成失败'));

    await expect(generateObject({
      model: 'mockModel',
      schema: recipeSchema,
      prompt: '生成一个无效的食谱',
    })).rejects.toThrow('生成失败');
  });
});
